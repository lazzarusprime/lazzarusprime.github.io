// requestbot.js const tmi = require('tmi.js'); const admin = require('firebase-admin'); const fs = require('fs'); const path = require('path'); // Load songs.json const songsPath = path.join(__dirname, 'songs.json'); const songsData = JSON.parse(fs.readFileSync(songsPath, 'utf8')); // Create lookup maps const songsById = {}; const songsByTitleArtist = {}; songsData.forEach(s => { songsById[s.id] = s; songsByTitleArtist[`${s.artist.toLowerCase()}|${s.song.toLowerCase()}`] = s; }); // Firebase setup const serviceAccount = require('./firebase-service-account.json'); admin.initializeApp({ credential: admin.credential.cert(serviceAccount), databaseURL: "https://rocksmith-requests-default-rtdb.firebaseio.com" }); const db = admin.database(); // Twitch setup const client = new tmi.Client({ options: { debug: true }, connection: { reconnect: true }, identity: { username: 'YourBotUsername', password: 'oauth:rqi3rxz2terwtvxy3wmjlcfgq6pul4' }, channels: [ 'YourChannel' ] }); client.connect(); client.on('message', async (channel, tags, message, self) => { if (self) return; let song = null; // RSPlaylist command: !sr song:ID const srMatch = message.match(/^!sr song:(\d+)$/i); if (srMatch) { const songId = parseInt(srMatch[1], 10); song = songsById[songId]; if (!song) { client.say(channel, `Song ID ${songId} not found.`); return; } } // Library command: !request "Artist" "Song" const reqMatch = message.match(/^!request\s+"(.+)"\s+"(.+)"$/i); if (reqMatch) { const artist = reqMatch[1].trim(); const title = reqMatch[2].trim(); const key = `${artist.toLowerCase()}|${title.toLowerCase()}`; const entry = songsByTitleArtist[key]; song = entry ? entry : { artist, song: title, owned: false }; } if (!song) return; // Ensure owned is boolean const owned = !!song.owned; // Push to Firebase with timestamp const newEntry = { artist: song.artist, song: song.song, owned: owned, timestamp: Date.now() }; db.ref('queue').push(newEntry, err => { if(err) console.error('Firebase push error:', err); }); });
